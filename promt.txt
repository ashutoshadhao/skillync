## SUPABASE DATABASE SCHEMA
Create these tables with RLS (Row Level Security) enabled:

-- Users (synced from Clerk webhook)
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  clerk_id TEXT UNIQUE NOT NULL,
  name TEXT,
  email TEXT,
  plan TEXT DEFAULT 'free' CHECK (plan IN ('free', 'pro', 'team')),
  razorpay_customer_id TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Resumes
CREATE TABLE resumes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  label TEXT DEFAULT 'My Resume',
  raw_text TEXT,
  parsed_skills TEXT[],
  parsed_titles TEXT[],
  experience_years INT,
  file_url TEXT,
  is_active BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Job Listings
CREATE TABLE job_listings (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  company TEXT NOT NULL,
  location TEXT,
  description TEXT,
  source_url TEXT UNIQUE,
  source_platform TEXT,
  salary_min INT,
  salary_max INT,
  is_remote BOOLEAN DEFAULT false,
  is_ghost BOOLEAN DEFAULT false,
  posted_at TIMESTAMPTZ,
  fetched_at TIMESTAMPTZ DEFAULT now()
);

-- Job Matches (per user)
CREATE TABLE job_matches (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  job_id UUID REFERENCES job_listings(id) ON DELETE CASCADE,
  resume_id UUID REFERENCES resumes(id),
  match_score INT,
  matched_skills TEXT[],
  missing_skills TEXT[],
  reasoning TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  UNIQUE(user_id, job_id)
);

-- Applications
CREATE TABLE applications (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  job_id UUID REFERENCES job_listings(id),
  status TEXT DEFAULT 'saved' CHECK (status IN ('saved','applied','viewed','interview','offer','rejected')),
  notes TEXT,
  applied_at TIMESTAMPTZ,
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Alerts
CREATE TABLE alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  keywords TEXT[],
  location TEXT,
  salary_min INT,
  remote_only BOOLEAN DEFAULT false,
  frequency TEXT DEFAULT 'daily' CHECK (frequency IN ('daily', 'instant')),
  is_active BOOLEAN DEFAULT true
);

-- Email Config (user's own SMTP)
CREATE TABLE email_configs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE UNIQUE,
  smtp_host TEXT,
  smtp_port INT DEFAULT 587,
  smtp_user TEXT,
  smtp_pass TEXT, -- store encrypted
  from_email TEXT,
  is_verified BOOLEAN DEFAULT false
);

-- Company Watchers
CREATE TABLE company_watchers (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  company_name TEXT,
  job_count_last_week INT DEFAULT 0,
  job_count_this_week INT DEFAULT 0,
  is_surge BOOLEAN DEFAULT false,
  last_checked TIMESTAMPTZ DEFAULT now()
);

## NODEMAILER — USER EMAIL SETUP
Users connect their own Gmail or SMTP in /settings/email:

// lib/mailer.ts
import nodemailer from 'nodemailer';

export async function createTransport(config: EmailConfig) {
  return nodemailer.createTransport({
    host: config.smtp_host,
    port: config.smtp_port,
    secure: false,
    auth: {
      user: config.smtp_user,
      pass: config.smtp_pass,
    },
  });
}

export async function sendDailyDigest(config, userEmail, matches) {
  const transport = await createTransport(config);
  await transport.sendMail({
    from: config.from_email,
    to: userEmail,
    subject: `Your top ${matches.length} job matches today — Skillync`,
    html: generateDigestHTML(matches), // build nice HTML email template
  });
}

// lib/scraper.ts — fetch from multiple sources

Sources to scrape:
1. Indeed India: https://indeed.com/jobs?q={keyword}&l={location}
2. Naukri: https://naukri.com/{keyword}-jobs
3. LinkedIn (public): https://linkedin.com/jobs/search?keywords={keyword}
4. Internshala (for freshers)
5. Wellfound/AngelList (for startups)

Use:
- axios + cheerio for HTML scraping
- Rotate user agents to avoid blocks
- Cache results in Supabase for 6 hours (don't re-fetch same URL)
- Run via /api/cron/fetch-jobs (Vercel cron every 6 hours)

Ghost job detection:
- Flag job as ghost if: posted_at < 45 days ago AND source_url still returns 200
- Add is_ghost: true to listing

Deduplication:
- Hash: company + title + location → check if exists before insert
